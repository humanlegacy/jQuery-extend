Hello-ui

jQuery.hello-ui.js 前端组件库

概况

部分组件依赖于其他组件库，具体请在实例页面查看页面源代码。

/*目录结构*/
hello-ui
    ├ libs       //组件依赖库
    ├ plugins       //插件扩展目录
    ├ hello-ui.css       //组件样式文件
    ├ jquery.hello-ui.js       //组件主程序,所有内置方法与插件可通过此程序调用

内置方法

为日常所需提供内置方法

内置方法 使用说明
$.hello('getPlug',['accordion']); 加载插件 传递字符串或数组
$.hello('getPlugName');获取已加载的扩展方法
$.hello('tools');获取Hello-ui所有内置方法
var changeToArray =$.hello('changeToArray',{name:'Peter',sex:'boy',age:32});对象转数组
var randomArray = $.hello('randomArray',["red", "blue", "yellow", "green"]);随机获取数组中某一项
var random = $.hello('random',[1,5]);生成指定范围内的随机数
var mOfArray = $.hello('mOfArray',[11,5,8,0,-3,7]);数组中的最大值和最小值
var removeOneVal = $.hello('removeOneVal',["red", "blue", "yellow", "green"],'green');删除数组中指定元素
var randomStr = $.hello('randomStr',5); 随机生成指定长度的数字字母组合
var color = $.hello('randomColor'); 随机色
$("button").click(function(){ var pos = $.hello('mousePos',$(this)); }); 获取鼠标在元素中的位置

扩展

在hello-ui基础上自定义扩展插件

;(function($) {
    $.helloPlug.prototype.getColor = function(param){
        //这里是你的代码···
    }
})(jQuery);

$.helloPlug.prototype.getColor = function(param){ 中

getColor 为你的插件名称

param 为传递的参数

//调用您自定义扩展插件
$.hello('getColor',param);

就是这么简单。

高级使用

在您的页面中引入hello-ui.css 与 jquery.hello-ui.js 即可随意调用组件中的插件与内置方法

$.hello('getPlug',['accordion']);    
//调用accordion插件，传递字符串与数组

手风琴

一个优雅的折叠伸缩插件，支持回调函数

$.hello('accordion',{
    event:'click',    // 事件
    speed: 200,    //响应速度
    callBack: function() {}     //执行后的回调函数
});

无缝轮播

无缝轮播，全屏，自定义尺寸，自动轮播，支持回调函数

$.hello('carousel',{
    ele:'.hello-carousel',    //绑定元素
    size: 'fullScreen',  //可选参数[700,350]指定宽高或全屏展示
    startIndex: 0,    //默认展示的轮播索引值
    horVertical: true,    //中间内容是否水平垂直居中
    arrow: true,    //是否显示左右导航箭头
    arrowText: ["Prev", "Next"],    //左右导航箭头的显示文本
    dots: true,    //是否显示焦点
    autoPlay: false,    //是否自动轮播
    interval: 3000,    //自动轮播时间间隔，毫秒
    speed:500,    //轮播切换的过渡时间
    onLeave: function(index) {},    //轮播离开时的回调函数
    onEnter: function(index) {}     //轮播进入时的回调函数
});

弹窗

支持拖拽，Ajax,自定义宽度，回调函数

$.hello('accordion',{
    type:'msg',    //弹窗场景，可选参数msg，confirm，loader，page
    title:'hello-ui dialog',    //弹窗标题
    width:500,    //弹窗宽度
    height:500,    //弹窗高度
    draggable:true,    //是否可拖拽
    msg:'老司机！',    //场景为msg时可用，显示提示消息
    url:'http://www.baidu.com',     //场景为loader或page可用,加载页面内容到窗体
    async:false,    //场景为loader或page可用,同步或异步
    method:'post',    //场景为loader或page可用,请求类型
    loadBeforeText:'内容加载中，请稍后...',    //场景为loader或page可用,请求前的等待提示
    cancelBtnTxt:'关闭',    //取消按钮文本
    confirmBtnTxt:'确定',    //确定按钮文本
    confirmFn:function(){},    //点击确认后的回调函数    
    cancelFn:function(){}     //点击取消后的回调函数
});

小提示

支持多场景，多位置展示，可执行回调函数的桌面提醒

$.hello('accordion',{        
    width:230,    //提示宽度
    position:'center',    //在不同位置展示，可选参数leftTop，leftBottom，rightTop，rightBottom，center
    type:'success',    //展示不同场景，default，success，error，warning，info
    msg:'msg!',    //提示的消息内容
    keep:3,    //提示持续时间，秒
    callBack: function() {}    //提示执行完成后的回调函数
});

下拉菜单

自定义下拉菜单，整合全国三级联动

$.hello('selecter',{
    ele:'.selecter',    //绑定元素
    title: '请选择',    //下拉菜单标题
    city: false,    //是否执行全国三级联动
    split: '-',    //城市分隔符号
    data: 'libs/data.json',    //城市数据源
    input: '#id'    //将选中的内容放入指定容器中
});

流程提醒

});

选项卡

支持自动轮播，多事件触发

$.hello('tabs',{
    ele:'.tabs',    //绑定元素
    triggerName: ['Black', 'Red', 'Yellow', 'White', 'Green'],    //选项卡触发元素

    event: 'mouseover'    //触发事件，可选参数mouseover，click
});

自动完成

填写表单时的提示输入,支持回调函数，多事件触发

$.hello('autoComplete',{
    ele:'.autoComplete',    //绑定元素
    data: [],    //数据源，其中可以是多个键值对对象
    autoFocus: true,    //是否自动触发或输入时触发下拉
    menu: '.demo-menu',     //为其绑定一个指定下拉,程序自动添加标签
    callBack: function(data) {}     //选择后的回调，data为一个对象，可获得点击元素的文本和索引
});

纯前端裁剪图片

基于canvas，可设定裁剪任意尺寸图片

$.hello('clip',{
    ele:'.clip',    //绑定元素
    width: 150,    //裁剪图片的宽度
    height: 150,    //裁剪图片的高度
    saveFileName: 'hello-clip.png',    //保存的文件名
    uploadUrl: "upload.php",    //图片上传服务器的地址
    uploadType: 'post',    //上传类型
    formChange: function() {    //上传完成后返回前端页面进行处理
        alert("上传图片中");
        $(".hello-clip-wrap").prepend("<img src='img/hello-clip.jpg' /≶"); //插入等待裁剪图片元素
        $(".helloClip-path").val("hello-ui/img/hello-clip.jpg"); //上传后返回的服务器图片地址，等待裁剪
    }
});

验证码

基于canvas，实现验证码功能

$.hello('clip',{
    ele:'.hello-captcha',    //绑定元素
    width:70,    //验证码宽度
    height: 25,    //验证码高度
    codeSize:4,    //验证码随机生成的长度,由数字字母组成
    fontColor:'random',     //验证码颜色随机或指定颜色
    font:'25px arial',    //验证码字体和大小
    bgLines:[5,8]     //验证码背景混淆的直线数量,5-8之间
});

简约滚动条

一个自定义滚动条，带回调函数

$.hello('clip',{
    step: 10,    //鼠标滚动一次滚轴时内容滑动的步长
    animate: null,    //滚动的动画,此时需要引入动画库
    duration: 200,    //滚动的动画过渡,毫秒
    barStyle:{},    //滚动条自定义样式
    scrollEnd: function() {}     //滚动至底部时的回调函数
});
